    \coordinate (Delta) at ({-sqrt(2)}, 0);

    \filldraw (Gamma) circle (1.2pt);
    \filldraw (Delta) circle (1.2pt);         

    % Draw the shape
    \draw[thick] (A) -- (B) -- (Gamma) -- (Delta) -- cycle;

    % Label the vertices
    \node[outer sep=2pt, above] at (A) {$A$};
    \node[outer sep=2pt, right] at (B) {$B$};
    \node[outer sep=2pt, below] at (Gamma) {$\Gamma$};
    \node[left] at (Delta) {$\Delta$};
