\documentclass[border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary {patterns,patterns.meta}
\usepackage{amsmath}

\pgfdeclarepattern{
  name=hatch,
  parameters={\hatchsize,\hatchangle,\hatchlinewidth},
  bottom left={\pgfpoint{-.05pt}{-.05pt}},
  top right={\pgfpoint{\hatchsize+.05pt}{\hatchsize+.05pt}},
  tile size={\pgfpoint{\hatchsize}{\hatchsize}},
  tile transformation={\pgftransformrotate{\hatchangle}},
  code={
    \pgfsetlinewidth{\hatchlinewidth}
    \pgfpathmoveto{\pgfpoint{-.05pt}{-.05pt}}
    \pgfpathlineto{\pgfpoint{\hatchsize+.05pt}{\hatchsize+.05pt}}
    \pgfpathmoveto{\pgfpoint{-.05pt}{\hatchsize+.05pt}}
    \pgfpathlineto{\pgfpoint{\hatchsize+.05pt}{-.05pt}}
    \pgfusepath{stroke}
  }
}

\tikzset{
  hatch size/.store in=\hatchsize,
  hatch angle/.store in=\hatchangle,
  hatch line width/.store in=\hatchlinewidth,
  hatch size=3pt,
  hatch angle=0pt,
  hatch line width=.5pt,
}
